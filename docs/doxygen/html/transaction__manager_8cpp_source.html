161bool TransactionManager::acquire_lock(TransactionId txn_id,
162 const std::string &resource,
163 LockType lock_type, bool wait) {
164 std::unique_lock<std::mutex> lock(mutex_);
165
166 // 检查事务是否存在
167 auto txn_it = transactions_.find(txn_id);
168 if (txn_it == transactions_.end() ||
169 txn_it->second.state != TransactionState::ACTIVE) {
170 return false;
171 }
172
173 // 检查是否已经持有该资源的锁
174 if (lock_table_.count(resource)) {
175 auto &locks = lock_table_[resource];
176 for (const auto &lock_entry : locks) {
177 if (lock_entry.txn_id == txn_id) {
178 // 已经持有更高级别的锁
179 if (lock_entry.type == LockType::EXCLUSIVE &&
180 lock_type == LockType::SHARED) {
181 return true;
182 }
183 // 已经持有相同类型的锁
184 if (lock_entry.type == lock_type) {
185 return true;
186 }
187 }
188 }
189 }
190
191 // 详细的锁获取逻辑
192 if (lock_type == LockType::SHARED) {
193 // 共享锁兼容性：只有排他锁不兼容
194 bool has_exclusive_lock = false;
195 if (lock_table_.find(resource) != lock_table_.end()) {
196 for (const auto &lock_entry : lock_table_[resource]) {
197 if (lock_entry.type == LockType::EXCLUSIVE) {
198 has_exclusive_lock = true;
199 if (wait) {
200 wait_graph_[txn_id].insert(lock_entry.txn_id);
201 }
202 }
203 }
204 }
205
206 if (!has_exclusive_lock) {
207 // 添加共享锁
208 LockEntry new_lock;
209 new_lock.txn_id = txn_id;
210 new_lock.resource = resource;
211 new_lock.type = LockType::SHARED;
212 new_lock.acquired_time = std::chrono::system_clock::now();
213 lock_table_[resource].push_back(new_lock);
214 std::cout << "Transaction " << txn_id
215 << " acquired shared lock on resource '" << resource << "'"
216 << std::endl;
217 return true;
218 }
219 return false;
220 } else if (lock_type == LockType::EXCLUSIVE) {
221 // 排他锁检查：不能有任何其他锁
222 if (lock_table_.find(resource) != lock_table_.end() && !lock_table_[resource].empty()) {
223 if (wait) {
224 for (const auto &lock_entry : lock_table_[resource]) {
225 wait_graph_[txn_id].insert(lock_entry.txn_id);
226 }
227 }
228 return false;
229 }
230
231 // 添加排他锁
232 LockEntry new_lock;
233 new_lock.txn_id = txn_id;
234 new_lock.resource = resource;
235 new_lock.type = LockType::EXCLUSIVE;
236 new_lock.acquired_time = std::chrono::system_clock::now();
237 lock_table_[resource].push_back(new_lock);
238 std::cout << "Transaction " << txn_id
239 << " acquired exclusive lock on resource '" << resource << "'"
240 << std::endl;
241 return true;
242 }
243
244 return false;

247void TransactionManager::release_lock(TransactionId txn_id,
248 const std::string &resource) {
249 std::unique_lock<std::mutex> lock(mutex_);
250
251 // 检查资源是否有锁
252 if (lock_table_.find(resource) != lock_table_.end()) {
253 auto &locks = lock_table_[resource];
254
255 // 查找并移除该事务对该资源的锁
256 auto it = std::remove_if(
257 locks.begin(), locks.end(),
258 [txn_id](const LockEntry &entry) { return entry.txn_id == txn_id; });
259
260 if (it != locks.end()) {
261 locks.erase(it, locks.end());
262
263 // 如果资源没有锁了，从锁表中删除
264 if (locks.empty()) {
265 lock_table_.erase(resource);
266 }
267 }
268 }
269
270 // 清理等待图
271 wait_graph_.erase(txn_id);
272 for (auto &[tid, waiting_for] : wait_graph_) {
273 waiting_for.erase(txn_id);
274 }
275

277 << resource << "'" << std::endl;

280bool TransactionManager::detect_deadlock(TransactionId txn_id) {
281 // 简化实现：使用拓扑排序检测死锁环
282 std::unordered_map<TransactionId, int> indegree;
283 std::queue<TransactionId> q;
284
285 // 初始化入度
286 for (const auto &[tid, _] : wait_graph_) {
287 indegree[tid] = 0;
288 }
289
290 for (const auto &[tid, dependents] : wait_graph_) {
291 for (TransactionId dependent : dependents) {
292 indegree[dependent]++;
293 }
294 }
295
296 // 将入度为0的节点加入队列
297 for (const auto &[tid, degree] : indegree) {
298 if (degree == 0) {
299 q.push(tid);
300 }
301 }
302
303 // 拓扑排序
304 while (!q.empty()) {
305 TransactionId current = q.front();
306 q.pop();
307
308 if (wait_graph_.count(current)) {
309 for (TransactionId dependent : wait_graph_[current]) {
310 indegree[dependent]--;
311 if (indegree[dependent] == 0) {
312 q.push(dependent);
313 }
314 }
315 }
316 }
317
318 // 检查是否存在环（即入度不为0的节点）
319 for (const auto &[tid, degree] : indegree) {
320 if (degree > 0) {
321 std::cout << "Deadlock detected involving transaction " << txn_id
322 << std::endl;
323 return true;
324 }
325 }
326
327 return false;
328}

391bool TransactionManager::can_acquire_lock(TransactionId txn_id,
392 const std::string &resource,
393 LockType lock_type) const {
394 // 检查资源是否有锁
395 if (lock_table_.find(resource) != lock_table_.end()) {
396 for (const auto &entry : lock_table_.at(resource)) {
397 // 如果是同一个事务，不需要检查
398 if (entry.txn_id == txn_id) {
399 continue;
400 }
401
402 // 如果请求的是排他锁，但资源已经有锁，无法获取
403 if (lock_type == LockType::EXCLUSIVE) {
404 return false;
405 }
406
407 // 如果请求的是共享锁，但资源已经有排他锁，无法获取
408 if (lock_type == LockType::SHARED && entry.type == LockType::EXCLUSIVE) {
409 return false;
410 }
411 }
412 }
413
414 return true;
