594 DiskManager disk_manager(test_db_file_, config);

596 // 创建多个线程进行读写操作
597 const int num_threads = 4;
598 const int num_pages = 10;
599 std::vector<std::thread> threads;
600
601 for (int i = 0; i < num_threads; ++i) {
602 threads.emplace_back([&, i]() {
603 // 每个线程写入和读取不同的页面
604 for (int j = 0; j < num_pages; ++j) {
605 int32_t page_id = i * num_pages + j;
606
607 // 创建测试数据
608 char write_data[8192];
609 CreateTestPageData(write_data, 8192, i + j);

611 // 写入页面
612 ASSERT_TRUE(disk_manager.WritePage(page_id, write_data));
613
614 // 读取页面

616 ASSERT_TRUE(disk_manager.ReadPage(page_id, read_data));
617
618 // 验证数据
619 ASSERT_TRUE(VerifyPageData(read_data, 8192, i + j));
620 }
621 });
622 }
