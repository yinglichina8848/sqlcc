248int32_t BufferPool::ReplacePage() {
249 SQLCC_LOG_DEBUG("Replacing page using LRU algorithm");
250
251 // 使用LRU算法查找最久未使用的页面
252 while (!lru_list_.empty()) {
253 // 获取LRU链表尾部的页面ID（最久未使用）
254 int32_t page_id = lru_list_.back();
255
256 // 检查页面引用计数
257 auto ref_it = page_refs_.find(page_id);
258 if (ref_it != page_refs_.end() && ref_it->second > 0) {
259 // 页面正在被使用，不能替换，移到头部
260 SQLCC_LOG_DEBUG("Page ID " + std::to_string(page_id) + " is in use, moving to head of LRU list");
261 MoveToHead(page_id);
262 continue;
263 }
264
265 // 检查是否为脏页，如果是则需要刷新到磁盘
266 auto dirty_it = dirty_pages_.find(page_id);
267 if (dirty_it != dirty_pages_.end() && dirty_it->second) {
268 auto page_it = page_table_.find(page_id);
269 if (page_it != page_table_.end()) {
270 // 刷新脏页到磁盘
271 SQLCC_LOG_DEBUG("Flushing dirty page ID " + std::to_string(page_id) + " to disk before replacement");
272 if (!disk_manager_->WritePage(*page_it->second)) {
273 std::string error_msg = "Failed to write dirty page ID " + std::to_string(page_id) + " to disk";
274 SQLCC_LOG_ERROR(error_msg);
275 // 无法刷新脏页，不能替换此页面
276 continue;
277 }
278 }
279 }
280
281 // 从缓冲池中移除页面
282 page_table_.erase(page_id);
283
284 // 从脏页表中移除
285 if (dirty_it != dirty_pages_.end()) {
286 dirty_pages_.erase(dirty_it);
287 }
288
289 // 从引用计数表中移除
290 if (ref_it != page_refs_.end()) {
291 page_refs_.erase(ref_it);
292 }
293
294 // 从LRU链表中移除
295 lru_list_.pop_back();
296 lru_map_.erase(page_id);
297
298 SQLCC_LOG_DEBUG("Page ID " + std::to_string(page_id) + " replaced");
299 return page_id;
300 }
301
302 // 无法找到可替换的页面
303 SQLCC_LOG_WARN("No page can be replaced in buffer pool");
304 return -1;
305}
