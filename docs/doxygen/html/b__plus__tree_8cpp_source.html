140BPlusTreeNode::BPlusTreeNode(StorageEngine *storage_engine, int32_t page_id,
141 bool is_leaf)
142 : storage_engine_(storage_engine), page_id_(page_id), parent_page_id_(-1),
143 is_leaf_(is_leaf), page_(nullptr) {
144
145 // 获取页面对象用于数据存储
146 if (storage_engine_) {
147 page_ = storage_engine_->FetchPage(page_id);
148 }
149
150 SQLCC_LOG_DEBUG(std::string("Created B+Tree ") +
151 (is_leaf ? "leaf" : "internal") +
152 " node: page_id=" + std::to_string(page_id));
153}

155BPlusTreeNode::~BPlusTreeNode() {
