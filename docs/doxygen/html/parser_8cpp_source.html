56std::vector<std::unique_ptr<Statement>> Parser::parseStatements() {
57 std::vector<std::unique_ptr<Statement>> statements;
58
59 // 确保初始token已加载
60 if (currentToken_.getType() == Token::END_OF_INPUT && lexer_) {
61 consume();
62 }
63
64 // 循环解析语句，直到遇到文件结束符
65 while (currentToken_.getType() != Token::END_OF_INPUT) {
66 try {
67 // 尝试解析单个完整的语句（包括分号）
68 auto stmt = parseStatement();
69 if (stmt) {
70 // 检查是否有分号结束符，如果有则消费它
71 if (currentToken_.getType() == Token::SEMICOLON) {
72 consume();
73 }
74 statements.push_back(std::move(stmt));
75 }
76 } catch (const std::exception &e) {
77 // 如果解析出错，尝试跳过当前语句（消费到分号或结束符）
78 // 这是简单的错误恢复机制
79 while (currentToken_.getType() != Token::END_OF_INPUT &&
80 currentToken_.getType() != Token::SEMICOLON) {
81 consume();
82 }
83 // 消费分号
84 if (currentToken_.getType() == Token::SEMICOLON) {
85 consume();
86 }
87 }
88 }
89
90 return statements;
91}
