132std::string DataType::extractParams(const std::string &str, int &param1,
133 int &param2) {
134 std::string result = str;
135 param1 = 0;
136 param2 = 0;
137
138 // 查找参数
139 size_t openParen = str.find('(');
140 if (openParen != std::string::npos) {
141 size_t closeParen = str.find(')', openParen);
142 if (closeParen != std::string::npos) {
143 // 提取参数字符串
144 std::string params =
145 str.substr(openParen + 1, closeParen - openParen - 1);
146
147 // 分离第一个和第二个参数
148 size_t comma = params.find(',');
149 if (comma != std::string::npos) {
150 // 有两个参数
151 try {
152 param1 = std::stoi(params.substr(0, comma));
153 param2 = std::stoi(params.substr(comma + 1));
154 } catch (const std::exception &) {
155 throw DataTypeException("Invalid numeric parameters in type: " + str);
156 }
157 } else {
158 // 只有一个参数
159 try {
160 param1 = std::stoi(params);
161 } catch (const std::exception &) {
162 throw DataTypeException("Invalid numeric parameter in type: " + str);
163 }
164 }
165
166 // 移除参数部分得到基本类型
167 result = str.substr(0, openParen);
168 }
169 }
170
171 return result;
172}

283DataType::Value DataTypeUtils::stringToValue(const std::string &str,
284 DataType::Type type) {

288

295 return static_cast<int16_t>(std::stoi(str));
296 }
297 case DataType::BIGINT: {
298 return static_cast<int64_t>(std::stoll(str));
299 }
300 case DataType::DECIMAL:
301 case DataType::DOUBLE: {
302 return std::stod(str);
303 }
304 case DataType::CHAR:
305 case DataType::VARCHAR:
306 case DataType::TEXT: {
307 // 对于字符串类型，直接存储字符串值
308 return std::string(str);
309 }
310 case DataType::BOOLEAN: {
311 std::string upper = str;
312 std::transform(upper.begin(), upper.end(), upper.begin(), ::toupper);
313 if (upper == "TRUE" || upper == "1" || upper == "YES") {
314 return true;
315 } else if (upper == "FALSE" || upper == "0" || upper == "NO") {
316 return false;
317 } else {
318 throw DataTypeException("Invalid boolean value: " + str);
319 }
320 }
321 case DataType::DATE:
322 case DataType::TIME:
323 case DataType::TIMESTAMP: {
324 // 日期时间验证可以在这里添加
325 // 目前简化处理，信任输入
326 return std::string(str);
327 }
328 default: {
329 return std::monostate{};
330 }
331 }
332 } catch (const std::exception &e) {
333 throw DataTypeException("Failed to convert string '" + str + "' to type '" +
334 typeToString(type) + "': " + e.what());

338bool DataTypeUtils::areCompatible(DataType::Type from, DataType::Type to) {
339 // 相同类型总是兼容的

355 (isNumericType(from) && to == DataType::BOOLEAN)) {
356 return true;
357 }
358
359 return false;

362DataType::Value DataTypeUtils::convertValue(const DataType::Value &value,
363 DataType::Type fromType,
364 DataType::Type toType) {
365 if (fromType == toType) {
366 return value;
367 }
368
369 // 如果是NULL值，直接返回
370 if (std::holds_alternative<std::monostate>(value)) {
371 return value;
372 }
373
374 // 数字类型转换
375 if (isNumericType(fromType) && isNumericType(toType)) {
376 double numVal = extractNumericValue(value, fromType);
377 return convertToNumericType(numVal, toType);
378 }
379
380 // 布尔类型转换
381 if (fromType == DataType::BOOLEAN && isNumericType(toType)) {
382 bool boolVal = std::get<bool>(value);
383 return convertToNumericType(boolVal ? 1.0 : 0.0, toType);
384 }
385
386 if (isNumericType(fromType) && toType == DataType::BOOLEAN) {
387 double numVal = extractNumericValue(value, fromType);
388 return numVal != 0.0;
389 }
390
391 // 不支持的转换返回原值（错误处理应该在调用者处）
392 return value;
393}
