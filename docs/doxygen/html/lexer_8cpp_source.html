49void Lexer::skipComment() {
50 // 跳过单行注释
51 if (match("--")) {
52 while (!isEOF() && currentChar() != '\n') {
53 advance();
54 }
55 if (!isEOF()) {
56 advance();
57 line_++;
58 column_ = 1;
59 }
60 }
61 // 跳过多行注释
62 else if (match("/*")) {
63 advance(2);
64 while (!isEOF() && !match("*/")) {
65 if (currentChar() == '\n') {
66 line_++;
67 column_ = 1;
68 }
69 advance();
70 }
71 if (match("*/")) {
72 advance(2);
73 }
74 }

92Token Lexer::readNumber() {
93 int start = position_;
94 int startColumn = column_;
95
96 // 读取整数部分
97 while (!isEOF() && isDigit(currentChar())) {
98 advance();
99 column_++;
100 }
101
102 // 读取小数部分
103 if (!isEOF() && currentChar() == '.' && isDigit(input_[position_ + 1])) {
104 advance();
105 column_++;
106 while (!isEOF() && isDigit(currentChar())) {
107 advance();
108 column_++;
109 }
110 }
111
112 std::string number = input_.substr(start, position_ - start);
113
114 return Token(Token::NUMERIC_LITERAL, number, line_, startColumn);

117Token Lexer::readString() {
118 int start = position_ + 1; // 跳过引号
119 int startColumn = column_;
120 char quoteChar = currentChar();
121
122 advance();
123 column_++;
124
125 while (!isEOF() && currentChar() != quoteChar) {
126 // 处理转义字符
127 if (currentChar() == '\\' && position_ + 1 < input_.size()) {
128 advance();
129 column_++;
130 }
131 advance();
132 column_++;
133 }
134
135 if (isEOF()) {
136 // 未闭合的字符串
137 return Token(Token::INVALID_TOKEN, "Unclosed string", line_, startColumn);
138 }
139
140 advance(); // 跳过结束引号
141 column_++;
142
143 std::string value = input_.substr(start, position_ - start - 1);
144 return Token(Token::STRING_LITERAL, value, line_, startColumn);
145}

187bool Lexer::match(const std::string &str) const {
188 if (position_ + str.size() > input_.size()) {
189 return false;
190 }
191
192 for (size_t i = 0; i < str.size(); i++) {
193 if (input_[position_ + i] != str[i]) {
194 return false;
195 }
196 }

283Token Lexer::scanNextToken() {
284 // 跳过空白字符
285 skipWhitespace();
286
287 // 检查是否到达文件末尾
288 if (isEOF()) {
289 return Token(Token::END_OF_INPUT, "", line_, column_);
290 }
291
292 // 跳过注释
293 if (match("--") || match("/*")) {
294 skipComment();
295 return scanNextToken();
296 }
297
298 char c = currentChar();
299 int currentColumn = column_;
300
301 // 标识符或关键字
302 if (isIdentifierStart(c)) {
303 return readIdentifier();
304 }
305
306 // 数字字面量
307 if (isDigit(c)) {
308 return readNumber();
309 }
310
311 // 字符串字面量
312 if (c == '\'' || c == '"') {
313 return readString();
314 }
315
316 // 运算符和标点符号
317 switch (c) {
318 case '=':
319 advance();
320 column_++;
321 return Token(Token::OPERATOR_EQUAL, "=", line_, currentColumn);
322 case '!':
323 advance();
324 column_++;
325 if (match('=')) {
326 advance();
327 column_++;
328 return Token(Token::OPERATOR_NOT_EQUAL, "!=", line_, currentColumn);
329 }
330 return Token(Token::KEYWORD_NOT, "!", line_, currentColumn);
331 case '<':
332 advance();
333 column_++;
334 if (match('=')) {
335 advance();
336 column_++;
337 return Token(Token::OPERATOR_LESS_EQUAL, "<=", line_, currentColumn);
338 }
339 return Token(Token::OPERATOR_LESS, "<", line_, currentColumn);
340 case '>':
341 advance();
342 column_++;
343 if (match('=')) {
344 advance();
345 column_++;
346 return Token(Token::OPERATOR_GREATER_EQUAL, ">=", line_, currentColumn);
347 }
348 return Token(Token::OPERATOR_GREATER, ">", line_, currentColumn);
349 case '+':
350 advance();
351 column_++;
352 return Token(Token::OPERATOR_PLUS, "+", line_, currentColumn);
353 case '-':
354 advance();
355 column_++;
356 return Token(Token::OPERATOR_MINUS, "-", line_, currentColumn);
357 case '*':
358 advance();
359 column_++;
360 return Token(Token::OPERATOR_MULTIPLY, "*", line_, currentColumn);
361 case '/':
362 advance();
363 column_++;
364 return Token(Token::OPERATOR_DIVIDE, "/", line_, currentColumn);
365 case '%':
366 advance();
367 column_++;
368 return Token(Token::OPERATOR_MODULO, "%", line_, currentColumn);
369 case '(':
370 advance();
371 column_++;
372 return Token(Token::PUNCTUATION_LEFT_PAREN, "(", line_, currentColumn);
373 case ')':
374 advance();
375 column_++;
376 return Token(Token::PUNCTUATION_RIGHT_PAREN, ")", line_, currentColumn);
377 case ',':
378 advance();
379 column_++;
380 return Token(Token::PUNCTUATION_COMMA, ",", line_, currentColumn);
381 case ';':
382 advance();
383 column_++;
384 return Token(Token::PUNCTUATION_SEMICOLON, ";", line_, currentColumn);
385 case '.':
386 advance();
387 column_++;
388 return Token(Token::PUNCTUATION_DOT, ".", line_, currentColumn);
389 case ':':
390 advance();
391 column_++;
392 if (match(':')) {
393 advance();
394 column_++;
395 return Token(Token::PUNCTUATION_DOUBLE_COLON, "::", line_, currentColumn);
396 }
397 return Token(Token::PUNCTUATION_COLON, ":", line_, currentColumn);
398 default:
399 // 无效字符
400 advance();
401 column_++;
402 return Token(Token::INVALID_TOKEN, std::string(1, c), line_, currentColumn);
403 }
404}
