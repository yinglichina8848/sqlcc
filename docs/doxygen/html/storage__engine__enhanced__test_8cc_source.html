80 template<typename F>
81 bool TestWithTimeout(F func, int timeout_seconds = 5) {
82 std::atomic<bool> completed{false};
83 std::atomic<bool> exception_occurred{false};
84 std::string exception_message;
85
86 std::cout << "[DEBUG] TestWithTimeout: Starting test with timeout of " << timeout_seconds << " seconds" << std::endl;
87
88 auto future = std::async(std::launch::async, [&]() {
89 try {
90 func();
91 completed = true;
92 std::cout << "[DEBUG] TestWithTimeout: Test completed successfully" << std::endl;
93 } catch (const std::exception& e) {
94 completed = true;
95 exception_occurred = true;
96 exception_message = e.what();
97 std::cout << "[DEBUG] TestWithTimeout: Exception occurred: " << exception_message << std::endl;
98 } catch (...) {
99 completed = true;
100 exception_occurred = true;
101 std::cout << "[DEBUG] TestWithTimeout: Unknown exception occurred" << std::endl;
102 }
103 });
104
105 auto start_time = std::chrono::steady_clock::now();
106 auto status = future.wait_for(std::chrono::seconds(timeout_seconds));
107 auto end_time = std::chrono::steady_clock::now();
108 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end_time - start_time).count();
109
110 if (status == std::future_status::timeout) {
111 std::cout << "[DEBUG] TestWithTimeout: TIMEOUT after " << duration << "ms, possible deadlock detected" << std::endl;
112 return false; // 超时
113 }
114
115 std::cout << "[DEBUG] TestWithTimeout: Test finished in " << duration << "ms" << std::endl;
116 return completed; // 返回是否完成
117 }
118};
