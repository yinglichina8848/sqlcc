353 std::unique_ptr<BufferPool> small_buffer_pool;
354
355 // 启动一个独立线程执行缓冲池构造
356 std::thread construction_thread([&]() {
357 small_buffer_pool = std::make_unique<BufferPool>(disk_manager_.get(), 2, *config_manager_);
358 construction_completed = true;
359 });
360
361 // 等待最多5秒构造完成
362 auto start_time = std::chrono::steady_clock::now();
363 const auto timeout_duration = std::chrono::seconds(5);
364
365 while (!construction_completed && !construction_timed_out) {
366 auto elapsed = std::chrono::steady_clock::now() - start_time;
367 if (elapsed >= timeout_duration) {
368 construction_timed_out = true;
369 break;
370 }
371 std::this_thread::sleep_for(std::chrono::milliseconds(50));
372 }
373
374 if (construction_timed_out) {
375 construction_thread.detach();
376 FAIL() << "BufferPool构造函数超时：缓冲池构造在5秒内未完成，检测到可能的死锁";
377 }
378
379 construction_thread.join();
380 ASSERT_NE(small_buffer_pool, nullptr);
381 std::cout << "BufferPool构造成功，池大小: 2" << std::endl;
