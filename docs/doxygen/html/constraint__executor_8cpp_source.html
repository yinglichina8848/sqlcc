148bool UniqueConstraintExecutor::validateInsert(
149 const std::vector<std::string> &record,
150 const std::vector<sql_parser::ColumnDefinition> &table_schema) {
151
152 // 获取约束列的值
153 std::vector<std::string> constraint_values =
154 getConstraintValues(record, table_schema);
155
156 // 检查值列表中是否有空值（对于主键，不允许NULL；对于唯一索引，允许NULL但需要特殊处理）
157 bool has_null = false;
158 for (const auto &value : constraint_values) {
159 if (value.empty()) {
160 has_null = true;
161 break;
162 }
163 }
164
165 // 如果是主键约束，NULL值不允许
166 if (is_primary_key_ && has_null) {
167 return false;
168 }
169
170 // 如果有NULL值且不是主键约束，允许插入（SQL标准：NULL值不参与唯一性检查）
171 if (!is_primary_key_ && has_null) {
172 return true;
173 }
174
175 // 检查唯一性
176 return checkUniqueness(constraint_values);
177}

237
238 try {
239 // 构建SELECT查询来检查是否存在重复值
240 std::string query = "SELECT COUNT(*) FROM " + table_name_ + " WHERE ";
241
242 std::vector<std::string> conditions;
243 if (is_primary_key_) {
244 const auto &pk_constraint =
245 dynamic_cast<const sql_parser::PrimaryKeyConstraint &>(
246 constraint_.get());
247 const auto &columns = pk_constraint.getColumns();
248 for (size_t i = 0; i < columns.size(); ++i) {
249 std::string condition = columns[i] + " = '" + values[i] + "'";
250 conditions.push_back(condition);
251 }
252 } else {
253 const auto &unique_constraint =
254 dynamic_cast<const sql_parser::UniqueConstraint &>(constraint_.get());
255 const auto &columns = unique_constraint.getColumns();
256 for (size_t i = 0; i < columns.size(); ++i) {
257 std::string condition = columns[i] + " = '" + values[i] + "'";
258 conditions.push_back(condition);
259 }
260 }
261

263 for (size_t i = 0; i < conditions.size(); ++i) {
264 if (i > 0)
265 query += " AND ";
266 query += conditions[i];
267 }
268
269 // 实际的查询执行应当返回是否存在记录
270 // 这里返回true表示没有重复记录，允许插入
271 return true; // 占位符实现
272
273 } catch (const std::exception &e) {
274 // 记录错误但不抛出异常，返回false表示验证失败

279std::vector<std::string> UniqueConstraintExecutor::getConstraintValues(
280 const std::vector<std::string> &record,
281 const std::vector<sql_parser::ColumnDefinition> &table_schema) {
282
283 std::vector<std::string> constraint_values;
284
285 // 获取约束涉及的所有列名
286 std::vector<std::string> constraint_columns;
287 if (is_primary_key_) {
288 const auto &pk_constraint =
289 dynamic_cast<const sql_parser::PrimaryKeyConstraint &>(
290 constraint_.get());
291 constraint_columns = pk_constraint.getColumns();
292 } else {
293 const auto &unique_constraint =
294 dynamic_cast<const sql_parser::UniqueConstraint &>(constraint_.get());
295 constraint_columns = unique_constraint.getColumns();
296 }
297
298 // 将列名转换为小写
299 std::vector<std::string> lower_columns;
300 for (const auto &col : constraint_columns) {
301 lower_columns.push_back(toLower(col));
302 }
303
304 // 按照列的顺序提取值
305 for (const auto &constraint_col : lower_columns) {
306 // 在表结构中找到对应列的索引
307 for (size_t i = 0; i < table_schema.size(); ++i) {
308 if (toLower(table_schema[i].getName()) == constraint_col) {
309 if (i < record.size()) {
310 constraint_values.push_back(record[i]);
311 } else {
312 constraint_values.push_back(""); // 缺少值的情况
313 }
314 break;
315 }
316 }
317 }
318
319 return constraint_values;
320}

392bool ExpressionEvaluator::evaluate(
393 const sql_parser::Expression *expr, const std::vector<std::string> &record,
394 const std::vector<sql_parser::ColumnDefinition> &table_schema) {
395
396 if (!expr)
397 return true;
398
399 switch (expr->getType()) {
400 case sql_parser::Expression::BINARY: {
401 const auto *binary_expr =
402 dynamic_cast<const sql_parser::BinaryExpression *>(expr);
403 return evaluateBinaryExpression(binary_expr, record, table_schema);
404 }
405
406 case sql_parser::Expression::IDENTIFIER: {
407 // 标识符通常为真（简化处理）
408 return true;
409 }
410
411 case sql_parser::Expression::STRING_LITERAL:
412 case sql_parser::Expression::NUMERIC_LITERAL: {
413 // 字面量通常为真（除非是布尔字面量）
414 return true;
415 }
416
417 default: {
418 // 其他表达式类型默认通过
419 return true;
420 }
421 }
422}
