465 bool success = true;
466
467 // 批量读取页面
468 for (const auto& pair : page_pairs) {
469 int32_t page_id = pair.first;
470 char* data = pair.second;
471
472 // 计算页面偏移量
473 off_t offset = static_cast<off_t>(page_id) * PAGE_SIZE;
474
475 // 定位到页面位置
476 if (lseek(fd, offset, SEEK_SET) == -1) {
477 SQLCC_LOG_ERROR("Failed to seek to page " + std::to_string(page_id) + " during batch read");
478 success = false;
479 continue;
480 }
481
482 // 读取页面数据
483 ssize_t bytes_read = read(fd, data, PAGE_SIZE);
484 if (bytes_read == -1) {
485 SQLCC_LOG_ERROR("Failed to read page " + std::to_string(page_id) + " during batch read");
486 success = false;
487 continue;
488 } else if (bytes_read < static_cast<ssize_t>(PAGE_SIZE)) {
489 // 如果读取的字节数小于页面大小，可能是文件末尾，填充剩余部分为0
490 memset(data + bytes_read, 0, PAGE_SIZE - bytes_read);
491 }
492 }
493
494 close(fd);
495 return success;
496}

528bool DiskManager::BatchPrefetchPages(const std::vector<int32_t>& page_ids) {
529 // 添加锁定保护，避免递归锁定
530 std::lock_guard<std::recursive_timed_mutex> lock(io_mutex_);
531
532 if (page_ids.empty()) {
533 return false;
534 }
535
536 // 过滤有效的页面ID并排序
537 std::vector<int32_t> valid_pages;
538 for (int32_t page_id : page_ids) {
539 if (page_id >= 0) {
540 valid_pages.push_back(page_id);
541 }
542 }
543
544 if (valid_pages.empty()) {
545 return false;
546 }
547
548 // 按页面ID排序，优化磁盘访问模式
549 std::sort(valid_pages.begin(), valid_pages.end());
550
551 // 使用文件描述符进行批量预读
552 int fd = open(db_file_name_.c_str(), O_RDONLY);
553 if (fd == -1) {
554 SQLCC_LOG_ERROR("Failed to open database file for batch prefetch: " + db_file_name_);
555 return false;
556 }
557
558 bool success = true;
559
560 // 合并连续的页面范围，提高预读效率
561 for (size_t i = 0; i < valid_pages.size(); ) {
562 int32_t start_page = valid_pages[i];
563 int32_t end_page = start_page;
564
565 // 找到连续的页面范围
566 while (i + 1 < valid_pages.size() && valid_pages[i + 1] == end_page + 1) {
567 i++;
568 end_page++;
569 }
570
571 // 计算连续范围的偏移量和大小
572 off_t offset = static_cast<off_t>(start_page) * PAGE_SIZE;
573 off_t size = static_cast<off_t>(end_page - start_page + 1) * PAGE_SIZE;
574
575 // 使用posix_fadvise建议操作系统预读连续页面范围
576 int result = posix_fadvise(fd, offset, size, POSIX_FADV_WILLNEED);
577 if (result != 0) {
578 success = false;
579 }
580
581 i++; // 移动到下一个不连续的页面
582 }
583
584 close(fd);
585 return success;
586}

600// What: Sync方法将文件缓冲区的内容强制写入磁盘
