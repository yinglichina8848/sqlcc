22std::vector<std::string> splitSqlStatements(const std::string& sqlContent) {
23 std::vector<std::string> statements;
24 std::string currentStatement;
25 bool inMultiLineComment = false;
26
27 for (size_t i = 0; i < sqlContent.length(); ++i) {
28 // 处理多行注释 /* ... */
29 if (!inMultiLineComment && i < sqlContent.length() - 1 &&
30 sqlContent[i] == '/' && sqlContent[i+1] == '*') {
31 inMultiLineComment = true;
32 i++; // 跳过 '*'
33 continue;
34 }
35
36 if (inMultiLineComment) {
37 if (i < sqlContent.length() - 1 &&
38 sqlContent[i] == '*' && sqlContent[i+1] == '/') {
39 inMultiLineComment = false;
40 i++; // 跳过 '/'
41 }
42 continue;
43 }
44
45 // 处理单行注释 --
46 if (i < sqlContent.length() - 1 && sqlContent[i] == '-' && sqlContent[i+1] == '-') {
47 while (i < sqlContent.length() && sqlContent[i] != '\n') {
48 ++i;
49 }
50 continue;
51 }
52
53 currentStatement += sqlContent[i];
54
55 // 检测分号作为语句结束符
56 if (sqlContent[i] == ';') {
57 // 去除前后空白字符
58 size_t start = currentStatement.find_first_not_of(" \t\n\r");
59 size_t end = currentStatement.find_last_not_of(" \t\n\r");
60 if (start != std::string::npos && end != std::string::npos) {
61 statements.push_back(currentStatement.substr(start, end - start + 1));
62 }
63 currentStatement.clear();
64 }
65 }
66
67 return statements;
68}
