156void WorkerThread(size_t thread_id, const InsertTestConfig& config,
157 std::vector<double>& latencies,
158 std::atomic<size_t>& operations_completed) {
159 std::stringstream info_ss;
160 info_ss << "Worker thread " << thread_id << " started: config.name = " << config.name.c_str() << ", record_size = " << config.record_size;
161 SQLCC_LOG_INFO(info_ss.str());
162
163 // 为每个线程创建独立的随机数生成器
164 std::mt19937 thread_rng(rng_() + thread_id);
165
166 // 计算每个线程需要插入的记录数
167 size_t records_per_thread = config.insert_count / config.thread_count;
168 size_t start_id = thread_id * records_per_thread;
169 size_t end_id = (thread_id == config.thread_count - 1) ?
170 config.insert_count : (thread_id + 1) * records_per_thread;
171 size_t thread_insert_count = end_id - start_id;
172
173 // 记录成功插入的记录数
174 size_t successful_inserts = 0;
175
176 // 执行插入操作
177 for (size_t i = 0; i < thread_insert_count; ++i) {
178 // 生成记录ID
179 size_t record_id = start_id + i;
180
181 // 记录开始时间
182 auto start_time = std::chrono::high_resolution_clock::now();
183
184 // 执行真实SQL插入
185 bool success = ExecuteRealRecordInsert(record_id, config.record_size);
186
187 // 记录结束时间
188 auto end_time = std::chrono::high_resolution_clock::now();
189
190 // 计算延迟（毫秒）
191 double latency_ms = std::chrono::duration<double, std::milli>
192 (end_time - start_time).count();
193
194 // 记录延迟
195 latencies.push_back(latency_ms);
196
197 // 如果成功，增加计数
198 if (success) {
199 successful_inserts++;
200 }
201
202 // 增加已完成操作计数
203 operations_completed++;
204 }
205
206 std::stringstream info_ss_end;
207 info_ss_end << "Worker thread " << thread_id << " completed: " << successful_inserts << " successful inserts out of " << thread_insert_count << " attempts";
208 SQLCC_LOG_INFO(info_ss_end.str());
209}

253 bool ExecuteRealRecordInsert(size_t record_id, size_t record_size) {
254 try {
255 // 生成随机数据
256 std::string data(record_size, '0');
257 for (size_t i = 0; i < record_size; ++i) {
258 data[i] = 'A' + (rng_() % 26);
259 }
260
261 // 构建SQL插入语句
262 std::string insert_sql = "INSERT INTO " + std::string(kTestTable) +
263 " (id, data) VALUES (" +
264 std::to_string(record_id) + ", '" +
265 data + "')";
266
267 // 执行真实的SQL插入
268 std::string result = sql_executor_->Execute(insert_sql);
269 if (result.empty() || !sql_executor_->GetLastError().empty()) {
270 std::stringstream error_ss;
271 error_ss << "Failed to execute SQL insert for record " << record_id << ": " << sql_executor_->GetLastError().c_str();
272 SQLCC_LOG_ERROR(error_ss.str());
273 return false;
274 }
275
276 // 更新计数器
277 records_inserted_++;
278
279 return true;
280 } catch (const std::exception& e) {
281 std::stringstream error_ss;
282 error_ss << "Exception during record insert: " << e.what();
283 SQLCC_LOG_ERROR(error_ss.str());
