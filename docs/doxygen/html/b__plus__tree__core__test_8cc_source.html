415
416 try {
417 leaf_node = new BPlusTreeLeafNode(storage_engine_.get(), page_id);
418
419 // 模拟并发插入 (简化测试)
420 const int num_threads = 3;
421 const int operations_per_thread = 50;
422 std::vector<std::thread> threads;
423 std::atomic<int> success_count(0);
424
425 // 启动多个线程进行并发操作
426 for (int t = 0; t < num_threads; ++t) {
427 threads.emplace_back(
428 [t, operations_per_thread, &leaf_node, &success_count]() {
429 try {
430 for (int i = 0; i < operations_per_thread; ++i) {
431 int key = t * operations_per_thread + i;
432 IndexEntry entry(std::to_string(key), key, 0);
433
434 if (leaf_node->Insert(entry)) {
435 success_count++;
436 }
437 }
438 } catch (...) {
439 // 并发访问可能发生异常，记录但不失败
440 }
441 });
442 }
443
444 // 等待所有线程完成
445 for (auto &thread : threads) {
446 thread.join();
447 }
448
449 // 验证成功操作数量
