493bool ConfigManager::ParseConfigLine(const std::string& line, std::string& current_section) {
494 // 去除首尾空白字符
495 // Why: 需要去除行首尾的空白字符，便于后续解析
496 // What: 使用erase和find_first/last_not_of去除空白字符
497 // How: 先去除行首空白，再去除行尾空白
498 std::string trimmed_line = line;
499 trimmed_line.erase(0, trimmed_line.find_first_not_of(" \t"));
500 trimmed_line.erase(trimmed_line.find_last_not_of(" \t") + 1);
501
502 // 跳过空行和注释行
503 // Why: 空行和注释行不包含配置信息，需要跳过
504 // What: 检查行是否为空或以#开头
505 // How: 使用empty和下标访问检查行内容
506 if (trimmed_line.empty() || trimmed_line[0] == '#') {
507 return true;
508 }
509
510 // 处理节行 [section]
511 // Why: 需要识别并处理节行，更新当前节名
512 // What: 检查行是否以[开头和]结尾，提取节名
513 // How: 使用下标访问检查首尾字符，使用substr提取节名
514 if (trimmed_line[0] == '[' && trimmed_line.back() == ']') {
515 current_section = trimmed_line.substr(1, trimmed_line.length() - 2);
516 return true;
517 }
518
519 // 处理键值对行 key = value
520 // Why: 需要识别并处理键值对行，提取键和值
521 // What: 查找等号位置，分割键和值，去除空白字符
522 // How: 使用find查找等号，使用substr分割，使用erase去除空白
523 size_t equal_pos = trimmed_line.find('=');
524 if (equal_pos != std::string::npos) {
525 std::string key = trimmed_line.substr(0, equal_pos);
526 std::string value = trimmed_line.substr(equal_pos + 1);
527
528 // 去除键和值的空白字符
529 // Why: 键和值周围可能有空白字符，需要去除
530 // What: 使用erase和find_first/last_not_of去除空白字符
531 // How: 与去除行空白类似的方式处理键和值
532 key.erase(0, key.find_first_not_of(" \t"));
533 key.erase(key.find_last_not_of(" \t") + 1);
534 value.erase(0, value.find_first_not_of(" \t"));
535 value.erase(value.find_last_not_of(" \t") + 1);
536
537 // 构建完整键名 section.key
538 // Why: 需要将节名和参数名组合成完整键名
539 // What: 如果有当前节名，则组合成"section.key"格式
540 // How: 使用字符串拼接和条件判断
541 std::string full_key = current_section.empty() ? key : current_section + "." + key;
542
543 // 解析值类型并存储
544 // Why: 需要根据值的字符串表示解析为合适的类型
545 // What: 检查值是否为布尔值、浮点数或整数，进行相应转换
546 // How: 使用字符串比较和try-catch进行类型转换
547 if (value == "true" || value == "false") {
548 config_map_[full_key] = (value == "true");
549 } else if (value.find('.') != std::string::npos) {
550 try {
551 config_map_[full_key] = std::stod(value);
552 } catch (const std::exception&) {
553 config_map_[full_key] = value;
554 }
555 } else {
556 try {
557 config_map_[full_key] = std::stoi(value);
558 } catch (const std::exception&) {
559 config_map_[full_key] = value;
560 }
561 }
562
563 return true;
564 }
565
566 return false;
567}
