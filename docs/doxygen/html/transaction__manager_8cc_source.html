132void TransactionManager::release_all_locks(TransactionId txn_id) {
133 // 简单实现：遍历锁表，释放该事务持有的所有锁
134 std::vector<std::string> empty_resources;
135
136 for (auto& [resource, locks] : lock_table_) {
137 locks.erase(
138 std::remove_if(locks.begin(), locks.end(),
139 [txn_id](const LockEntry& entry) {
140 return entry.txn_id == txn_id;
141 }),
142 locks.end());
143
144 // 记录空资源锁列表，稍后删除
145 if (locks.empty()) {
146 empty_resources.push_back(resource);
147 }
148 }
149
150 // 在遍历完成后删除空资源锁列表，避免迭代器失效
151 for (const auto& resource : empty_resources) {
152 lock_table_.erase(resource);
153 }
154
155 // 清理等待图
156 wait_graph_.erase(txn_id);
157 for (auto& [tid, waiting_for] : wait_graph_) {
158 waiting_for.erase(txn_id);
159 }
160}

162// 实现acquire_lock方法
163bool TransactionManager::acquire_lock(TransactionId txn_id, const std::string& resource, LockType lock_type, bool wait) {
164 std::unique_lock<std::mutex> lock(mutex_);
165
166 // 检查事务是否存在且活跃
167 if (transactions_.find(txn_id) == transactions_.end() ||
168 transactions_[txn_id].state != TransactionState::ACTIVE) {
169 return false;
170 }
171
172 // 检查是否已经持有该资源的锁
173 if (lock_table_.contains(resource)) {
174 for (const auto& lock_entry : lock_table_[resource]) {
175 if (lock_entry.txn_id == txn_id) {
176 // 已经持有锁
177 if (lock_entry.type == lock_type) {
178 // 已持有相同类型的锁
179 return true;
180 }
181
182 // 尝试锁升级
183 if (lock_type == LockType::EXCLUSIVE && lock_entry.type == LockType::SHARED) {
184 // 检查是否可以升级（没有其他事务持有锁）
185 bool can_upgrade = true;
186 for (const auto& other_entry : lock_table_[resource]) {
187 if (other_entry.txn_id != txn_id) {
188 can_upgrade = false;
189 break;
190 }
191 }
192
193 if (can_upgrade) {
194 // 移除原锁
195 lock_table_[resource].erase(
196 std::remove_if(lock_table_[resource].begin(), lock_table_[resource].end(),
197 [txn_id](const LockEntry& entry) { return entry.txn_id == txn_id; }),
198 lock_table_[resource].end());
199
200 // 添加排他锁
201 LockEntry new_lock;
202 new_lock.txn_id = txn_id;
203 new_lock.resource = resource;
204 new_lock.type = LockType::EXCLUSIVE;
205 new_lock.acquired_time = std::chrono::system_clock::now();
206 lock_table_[resource].push_back(new_lock);
207 return true;
208 }
209
210 if (wait) {
211 // 记录等待关系
212 for (const auto& other_entry : lock_table_[resource]) {
213 if (other_entry.txn_id != txn_id) {
214 wait_graph_[txn_id].insert(other_entry.txn_id);
215 }
216 }
217 }
218 return false;
219 }
220
221 // 已持有更高级别的锁
222 if (lock_entry.type == LockType::EXCLUSIVE && lock_type == LockType::SHARED) {
223 return true;
224 }
225 }
226 }
227 }
228
229 // 新锁获取逻辑
230 if (lock_type == LockType::SHARED) {
231 // 共享锁兼容性：只有排他锁不兼容
232 bool has_exclusive_lock = false;
233 if (lock_table_.contains(resource)) {
234 for (const auto& lock_entry : lock_table_[resource]) {
235 if (lock_entry.type == LockType::EXCLUSIVE) {
236 has_exclusive_lock = true;
237 if (wait) {
238 wait_graph_[txn_id].insert(lock_entry.txn_id);
239 }
240 }
241 }
242 }
243
244 if (!has_exclusive_lock) {
245 // 添加共享锁
246 LockEntry new_lock;
247 new_lock.txn_id = txn_id;
248 new_lock.resource = resource;
249 new_lock.type = LockType::SHARED;
250 new_lock.acquired_time = std::chrono::system_clock::now();
251 lock_table_[resource].push_back(new_lock);
252 return true;
253 }
254 return false;
255 } else if (lock_type == LockType::EXCLUSIVE) {
256 // 排他锁检查：不能有任何其他锁
257 if (lock_table_.contains(resource) && !lock_table_[resource].empty()) {
258 if (wait) {
259 for (const auto& lock_entry : lock_table_[resource]) {
260 wait_graph_[txn_id].insert(lock_entry.txn_id);
261 }
262 }
263 return false;
264 }
265
266 // 添加排他锁
267 LockEntry new_lock;
268 new_lock.txn_id = txn_id;
269 new_lock.resource = resource;
270 new_lock.type = LockType::EXCLUSIVE;
271 new_lock.acquired_time = std::chrono::system_clock::now();
272 lock_table_[resource].push_back(new_lock);
273 return true;
274 }
275
276 return false;
277}

280void TransactionManager::release_lock(TransactionId txn_id, const std::string &resource) {
281 std::unique_lock<std::mutex> lock(mutex_);
282
283 // 检查资源是否有锁
284 if (lock_table_.contains(resource)) {
285 auto& locks = lock_table_[resource];
286
287 // 查找并移除该事务对该资源的锁
288 auto it = std::remove_if(locks.begin(), locks.end(),
289 [txn_id](const LockEntry& entry) {
290 return entry.txn_id == txn_id;
291 });
292
293 if (it != locks.end()) {
294 locks.erase(it, locks.end());
295
296 // 如果资源没有锁了，从锁表中删除
297 if (locks.empty()) {
298 lock_table_.erase(resource);
299 }
300 }
301 }
302
303 // 清理等待图
304 wait_graph_.erase(txn_id);
305 for (auto& [tid, waiting_for] : wait_graph_) {
306 waiting_for.erase(txn_id);
307 }
308}

327bool TransactionManager::can_acquire_lock(sqlcc::TransactionId txn_id, const std::string &resource, LockType lock_type) const {
328 // 检查资源是否有锁
329 if (lock_table_.contains(resource)) {
330 for (const auto& entry : lock_table_.at(resource)) {
331 // 如果是同一个事务，不需要检查
332 if (entry.txn_id == txn_id) {
333 continue;
334 }
335
336 // 如果请求的是排他锁，但资源已经有锁，无法获取
337 if (lock_type == LockType::EXCLUSIVE) {
338 return false;
339 }
340
341 // 如果请求的是共享锁，但资源已经有排他锁，无法获取
342 if (lock_type == LockType::SHARED && entry.type == LockType::EXCLUSIVE) {
343 return false;
344 }
345 }
346 }
347
348 return true;
349}
