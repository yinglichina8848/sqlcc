439TEST_F(ConfigManagerTest, ThreadSafety) {
440 ConfigManager& config = ConfigManager::GetInstance();
441
442 // 加载测试配置文件
443 config.LoadConfig(test_config_file_.string());
444
445 const int num_threads = 10;
446 const int num_operations = 100;
447 std::vector<std::thread> threads;
448
449 // 创建多个线程同时读写配置
450 for (int i = 0; i < num_threads; ++i) {
451 threads.emplace_back([&config, i, num_operations]() {
452 for (int j = 0; j < num_operations; ++j) {
453 // 写入配置
454 std::string key = "thread." + std::to_string(i) + ".key." + std::to_string(j);
455 config.SetValue(key, i * j);
456
457 // 读取配置
458 int value = config.GetInt("database.page_size", 0);
459 EXPECT_GT(value, 0);
460 }
461 });
462 }
463
464 // 等待所有线程完成
465 for (auto& thread : threads) {
466 thread.join();
467 }
468
469 // 验证配置值是否正确
470 for (int i = 0; i < num_threads; ++i) {
471 for (int j = 0; j < num_operations; ++j) {
472 std::string key = "thread." + std::to_string(i) + ".key." + std::to_string(j);
473 int value = config.GetInt(key, -1);
474 EXPECT_EQ(value, i * j);
475 }
476 }
