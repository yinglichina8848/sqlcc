207 void Log(LogLevel level, const std::string& message) {
208 if (level < log_level_) {
209 return;
210 }
211
212 auto now = std::chrono::system_clock::now();
213 auto time_t = std::chrono::system_clock::to_time_t(now);
214 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
215 now.time_since_epoch()) % 1000;
216
217 std::ostringstream timestamp;
218 timestamp << std::put_time(std::localtime(&time_t), "%Y-%m-%d %H:%M:%S");
219 timestamp << '.' << std::setfill('0') << std::setw(3) << ms.count();
220
221 std::string level_str;
222 switch (level) {
223 case LogLevel::DEBUG: level_str = "DEBUG"; break;
224 case LogLevel::INFO: level_str = "INFO"; break;
225 case LogLevel::WARN: level_str = "WARN"; break;
226 case LogLevel::ERROR: level_str = "ERROR"; break;
227 }
228
229 std::ostringstream log_msg;
230 log_msg << "[" << timestamp.str() << "] [" << level_str << "] " << message;
231
232 if (use_file_ && log_file_.is_open()) {
233 log_file_ << log_msg.str() << std::endl;
234 log_file_.flush();
235 } else {
236 if (level == LogLevel::ERROR) {
237 std::cerr << log_msg.str() << std::endl;
238 } else {
239 std::cout << log_msg.str() << std::endl;
240 }
241 }
242 }

244 LogLevel log_level_;
245 std::ofstream log_file_;
246 bool use_file_;
