117
118 // 格式化时间戳
119 std::ostringstream timestamp;
120 timestamp << std::put_time(std::localtime(&time_t), "%Y-%m-%d %H:%M:%S");
121 timestamp << '.' << std::setfill('0') << std::setw(3) << ms.count();
122
123 // 获取日志级别字符串
124 std::string level_str;
125 switch (level) {

129 case LogLevel::INFO:

132 case LogLevel::WARN:

138 }
139
140 // 构造完整日志消息
141 std::ostringstream log_msg;
142 log_msg << "[" << timestamp.str() << "] [" << level_str << "] " << message;
143
144 // 输出日志
145 if (use_file_ && log_file_.is_open()) {
146 log_file_ << log_msg.str() << std::endl;
147 log_file_.flush();
