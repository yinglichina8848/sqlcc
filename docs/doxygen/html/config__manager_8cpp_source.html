58// 解析配置行
59bool ConfigManager::ParseConfigLine(const std::string& line, std::string& current_section) {
60 // 简化的配置行解析
61 std::string trimmed = line;
62 // 移除首尾空格
63 trimmed.erase(0, trimmed.find_first_not_of(" \t"));
64 trimmed.erase(trimmed.find_last_not_of(" \t\r\n") + 1);
65
66 // 跳过空行和注释
67 if (trimmed.empty() || trimmed[0] == '#') {
68 return true;
69 }
70
71 // 处理section行 [section]
72 if (trimmed[0] == '[' && trimmed.back() == ']') {
73 current_section = trimmed.substr(1, trimmed.size() - 2);
74 return true;
75 }
76
77 // 处理key=value行
78 size_t eq_pos = trimmed.find('=');
79 if (eq_pos != std::string::npos) {
80 std::string key = trimmed.substr(0, eq_pos);
81 std::string value = trimmed.substr(eq_pos + 1);
82
83 // 移除键和值的空格
84 key.erase(key.find_last_not_of(" \t") + 1);
85 value.erase(0, value.find_first_not_of(" \t"));
86 value.erase(value.find_last_not_of(" \t\r\n") + 1);
87
88 // 如果有section，添加section前缀
89 if (!current_section.empty()) {
90 key = current_section + "." + key;
91 }
92
93 // 尝试解析不同类型的值
94 if (value == "true" || value == "false") {
95 SetValue(key, value == "true");
96 } else if (value.find_first_not_of("-0123456789") == std::string::npos) {
97 SetValue(key, std::stoi(value));
98 } else if (value.find_first_not_of("-0123456789.") == std::string::npos &&
99 value.find('.') != std::string::npos) {
100 SetValue(key, std::stod(value));
101 } else {
102 // 去掉引号（如果有）
103 if (value.size() >= 2 && ((value[0] == '"' && value.back() == '"') ||
104 (value[0] == '\'' && value.back() == '\''))) {
105 value = value.substr(1, value.size() - 2);
106 }
107 SetValue(key, value);
108 }
109
110 return true;
111 }
112
113 return false;
114}
